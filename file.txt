import os
import requests

def download_files(filename):
  """
  Downloads files listed in a text file.

  Args:
    filename: The name of the text file containing file URLs.
  """
  try:
    with open(filename, "r") as file:
      for line in file:
        # Remove trailing newline and whitespace
        url = line.strip()
        # Split URL and filename (optional, modify based on format)
        filename = url.split("/")[-1]
        
        # Download the file
        response = requests.get(url)
        response.raise_for_status()  # Raise exception for non-2XX status codes

        # Create download directory if not exists
        os.makedirs("downloads", exist_ok=True)

        # Save the file
        with open(f"downloads/{filename}", "wb") as f:
          f.write(response.content)
          print(f"Downloaded: {filename}")

  except FileNotFoundError:
    print(f"Error: File not found: {filename}")
  except requests.exceptions.RequestException as e:
    print(f"Error downloading {url}: {e}")

# Replace "urls.txt" with your actual file name
download_files("urls.txt")
